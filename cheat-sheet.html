SQL Queries

Cheat Sheet

Query SQL
Select all data SELECT * FROM table_name
Select certain column data SELECT
column_name1, column_name2
FROM table_name
Select certain column data and organize by column descending SELECT
column_name1, column_name2
FROM table_name
ORDER BY column_name2 DESC;
Select data if other data matches search query SELECT
column_name
FROM table_name
WHERE another_column=num/string
Select certain columns of data where columns matches search query (number greater than) and in ascending order SELECT
column_name1, column_name2
FROM table_name
WHERE column_name2>num
ORDER BY column_name2 ASC;
